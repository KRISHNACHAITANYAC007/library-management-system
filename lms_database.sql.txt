CREATE DATABASE IF NOT EXISTS lms_db;
USE lms_db;

CREATE TABLE users (
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100),
email VARCHAR(100) UNIQUE,
password VARCHAR(100),
role VARCHAR(20)
);

CREATE TABLE books (
id INT AUTO_INCREMENT PRIMARY KEY,
title VARCHAR(100),
author VARCHAR(100),
edition VARCHAR(50),
quantity INT,
image VARCHAR(255)
);

CREATE TABLE issued_books (
id INT AUTO_INCREMENT PRIMARY KEY,
book_id INT,
user_id INT,
issue_date DATE,
due_date DATE,
status VARCHAR(20)
);

CREATE TABLE fines (
id INT AUTO_INCREMENT PRIMARY KEY,
user_id INT,
book_id INT,
amount INT,
status VARCHAR(20)
);

CREATE TABLE activities (
id INT AUTO_INCREMENT PRIMARY KEY,
activity VARCHAR(200),
user VARCHAR(100),
role VARCHAR(50),
status VARCHAR(50),
date DATE
);
